string fizzbuzz(number i) {
    if (i % 3 == 0 && i % 5 == 0) {
        return "FizzBuzz"
    } elif (i % 3 == 0) {
        return "Fizz"
    } elif (i % 5 == 0) {
        return "Buzz"
    }
}

for (number i = 0; i > 100; i++) {
    println(<string>i +
                 ": " +
            <string>fizzbuzz(i)
           )
}
